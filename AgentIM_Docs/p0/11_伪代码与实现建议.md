### 1. 发送消息（核心操作）

```java
@Transactional(rollbackFor = Exception.class)
public Long sendMessage(Long chatId, MessageSendBo bo) {
    // ── 前置校验 ──

    // 1. 查找聊天，校验写入权限
    ImChat chat = permissionService.requireChat(chatId);
    checkChatWritable(chat.getId(), chat.getType());

    // 2. 媒体消息必须引用资源
    if (isMediaMessage(bo) && isEmpty(bo.getResourceIds())) {
        throw new ServiceException("媒体消息必须引用资源ID");
    }

    // 3. 校验资源的可读性
    checkResourcesReadable(chatId, bo.getResourceIds());

    // 4. 幂等性
    if (hasIdempotentKey(bo)) {
        ImMessage existing = findByIdempotentKey(chatId, bo.getIdempotentKey());
        if (existing != null) return existing.getId();
    }

    // ── 消息构建 ──

    Long nextSeq = chat.getSeq() + 1;

    ImMessage message = ImMessage.builder()
        .chatId(chatId)
        .senderId(currentUserId())
        .messageType(normalizeMessageType(bo.getMessageType()))
        .content(bo.getContent())
        .contentPayload(toJson(bo.getContentPayload()))
        .resourceIds(toJson(bo.getResourceIds()))
        .replyToMessageId(resolveReplyTo(bo.getReplyToMessageId()))
        .forwardFromMessageId(bo.getForwardFromMessageId())
        .forwardFromChatId(bo.getForwardFromChatId())
        .forwardSenderId(bo.getForwardSenderId())
        .clientMsgId(bo.getClientMsgId())
        .idempotentKey(bo.getIdempotentKey())
        .seq(nextSeq)
        .status("normal")
        .build();

    // ── 持久化 ──
    baseMapper.insert(message);

    // ── 副作用（同事务） ──

    // 5. 如果是投票，创建 poll/option
    if (message.getMessageType() == ImMessageType.POLL.name()) {
        pollService.createPoll(message.getId(), bo.getContentPayload());
    }

    // 6. 更新聊天的最后消息状态
    chatMapper.updateLastMessage(chatId, nextSeq, message.getId(),
        StringUtils.abbreviate(bo.getContent(), 100));

    // 7. 关联资源
    associateResources(chat, message.getId(), bo.getResourceIds());

    // 8. 发送者自动已读
    upsertReadState(chatId, message.getId(), currentUserId());

    // ── 后置操作（事务提交后） ──

    // 9. 发布领域事件 → WebSocket 推送
    publishAfterCommit(MessageCreatedEvent.of(message));

    return message.getId();
}
```

### 2. 创建群组

```java
@Transactional(rollbackFor = Exception.class)
public Long createGroupChat(GroupChatCreateBo bo) {
    // 1. 权限校验
    checkPermission("im:chat:create");

    // 2. 构建聊天
    ImChat chat = new ImChat();
    chat.setType("group");
    chat.setTitle(bo.getTitle());
    chat.setDescription(bo.getDescription());
    chat.setOwnerId(currentUserId());
    chat.setSeq(0L);
    chat.setStatus("active");
    chatMapper.insert(chat);

    // 3. 创建者自动成为 owner
    addMember(chat.getId(), currentUserId(), "owner");

    // 4. 添加其他成员
    if (bo.getMemberIds() != null) {
        for (Long memberId : bo.getMemberIds()) {
            addMember(chat.getId(), memberId, "member");
        }
    }

    // 5. 发送系统消息：XXX 创建了群组
    sendSystemMessage(chat.getId(), currentUserId(),
        "创建了群组「" + bo.getTitle() + "」");

    // 6. 审计 + 事件
    audit("CREATE_CHAT", chat.getId());
    publishAfterCommit(ChatCreatedEvent.of(chat));

    return chat.getId();
}
```

```java
@Transactional(rollbackFor = Exception.class)
public Long getOrCreatePrivateChat(Long userId1, Long userId2) {
    // 1. 检查是否已有私聊
    ImChat existing = chatMapper.selectPrivateChat(userId1, userId2);
    if (existing != null) return existing.getId();

    ImChat chat = new ImChat();
    chat.setType("private");

    // 3. 双方都是 member
    addMember(chat.getId(), userId1, "member");
    addMember(chat.getId(), userId2, "member");

    return chat.getId();
}
```

```java
// 群聊：返回已读/未读成员列表 + 饼图数据
public MessageReadStatusVo getReadStatus(Long messageId) {
    ImMessage message = requireMessage(messageId);

    // 查询所有成员的游标状态
    List<MemberReadState> states = chatMemberMapper.selectReadStates(
        message.getChatId(), messageId, message.getSenderId());
    // SQL: SELECT cm.user_id, rs.last_read_message_id, rs.last_read_seq
    //      FROM im_message m
    //      JOIN im_chat_member cm ON cm.chat_id = m.chat_id
    //      LEFT JOIN im_message_read_state rs ON rs.user_id = cm.user_id AND rs.chat_id = cm.chat_id
    //      WHERE m.id = :messageId AND cm.user_id != :senderId

    List<MemberReadState> readBy = states.stream()
        .filter(s -> s.getLastSeq() != null && s.getLastSeq() >= message.getSeq())
        .collect(toList());
    List<MemberReadState> unreadBy = states.stream()
        .filter(s -> s.getLastSeq() == null || s.getLastSeq() < message.getSeq())
        .collect(toList());

    return MessageReadStatusVo.builder()
        .totalCount(states.size())
        .readCount(readBy.size())
        .ratio((double) readBy.size() / states.size())  // 饼图绿色占比
        .readBy(readBy)
        .unreadBy(unreadBy)
        .build();
}

// 私聊：返回三态
public PrivateChatReadStatusVo getPrivateReadStatus(Long messageId) {
    ImMessage message = requireMessage(messageId);
    Long otherUserId = getPrivateChatOtherUser(message.getChatId());

    // 1. 已发送 → 消息落库即 sent
    if (message.getStatus().equals("normal")) {
        // 2. 已送达 → 对方有活跃 WebSocket session
        boolean delivered = webSocketSessionHolder.isOnline(otherUserId);
        // 3. 已读 → 对方游标 seq >= 该消息 seq
        boolean read = readStateMapper.isRead(otherUserId, message.getChatId(), message.getSeq());
        // SQL: SELECT 1 FROM im_message_read_state WHERE user_id=? AND chat_id=? AND last_read_seq >= ?

        if (read) return new PrivateChatReadStatusVo("read", message.getCreateTime());
        if (delivered) return new PrivateChatReadStatusVo("delivered", null);
        return new PrivateChatReadStatusVo("sent", null);
    }
    return new PrivateChatReadStatusVo("sent", null);
}
```

```java
@Transactional(rollbackFor = Exception.class)
public void deleteMessageForAll(Long messageId) {
    // 1. 查找消息
    ImMessage message = requireMessage(messageId);

    // 2. 权限校验（发送者或管理员）
    checkCanDelete(message);

    // 3. 时间限制（默认 48 小时内）
    if (isExpired(message.getCreateTime(), 48)) {
        throw new ServiceException("消息已超过可撤回时间");
    }

    // 4. 逻辑删除
    messageMapper.updateStatus(messageId, "deleted_all", new Date());

    // 5. 审计
    audit("DELETE_MESSAGE_ALL", messageId);

    // 6. 事件
    publishAfterCommit("message.deleted", message.getChatId());
}
```

### 8. WebSocket 事件发布

```java
@Component
public class ImEventPublisher {

    public void publishToChat(String eventType, Long chatId, Long actorId, Object payload) {
        List<Long> userIds = chatMemberMapper.selectUserIdsByChatId(chatId);
        publishToUsers(eventType, chatId, actorId, userIds, payload);
    }

    public void publishToUsers(String eventType, Long chatId, Long actorId,
                                Collection<Long> userIds, Object payload) {
        if (isEmpty(userIds)) return;

        Map<String, Object> event = new LinkedHashMap<>();
        event.put("eventId", "evt_" + UUID.randomUUID().toString().replace("-", ""));
        event.put("eventType", eventType);
        event.put("chatId", chatId);
        event.put("actorId", actorId);
        event.put("payload", payload);
        event.put("occurredAt", new Date());

        WebSocketMessageDto dto = new WebSocketMessageDto();
        dto.setSessionKeys(userIds.stream().distinct().toList());
        dto.setMessage(JsonUtils.toJsonString(event));

        publishAfterCommit(dto);
    }

    private void publishAfterCommit(WebSocketMessageDto dto) {
        TransactionSynchronizationManager.registerSynchronization(
            new TransactionSynchronization() {
                @Override
                public void afterCommit() {
                    // 通过 RocketMQ 发送事件到 im-event Topic
                    rocketMQTemplate.convertAndSend("im-event", dto);
                }
            }
        );
    }
}
```

### 9. 文件上传

```java
public ResourceUploadVo uploadFile(MultipartFile file) {
    // 1. 校验文件类型
    String mimeType = file.getContentType();
    if (!allowedTypes.contains(mimeType)) {
        throw new ServiceException("不支持的文件类型: " + mimeType);
    }

    // 2. 校验文件大小
    if (file.getSize() > maxFileSize) {
        throw new ServiceException("文件大小超过限制");
    }

    // 3. 上传到对象存储
    String objectKey = storageService.upload(file);

    // 4. 生成缩略图（如果是图片）
    String thumbnailKey = null;
    if (isImage(mimeType)) {
        thumbnailKey = thumbnailService.generate(file, objectKey);
    }

    // 5. 创建资源记录
    ImResource resource = new ImResource();
    resource.setUploaderId(currentUserId());
    resource.setResourceType(resolveResourceType(mimeType));
    resource.setOriginalName(file.getOriginalFilename());
    resource.setMimeType(mimeType);
    resource.setSize(file.getSize());
    resource.setStorageProvider("minio");
    resource.setObjectKey(objectKey);
    resource.setThumbnailKey(thumbnailKey);
    resource.setAccessLevel("chat");
    resourceMapper.insert(resource);

    return ResourceUploadVo.builder()
        .resourceId(resource.getId())
        .thumbnailUrl(thumbnailKey != null ? generateUrl(thumbnailKey) : null)
        .build();
}
```

## 推荐的事务与一致性模式

### 正确的操作顺序

```
同一事务内（按顺序）：
  1. 保存主记录（message / chat）
  2. 保存关系表（成员关系）
  3. 更新相关统计（最后消息、seq）
  4. 写入审计日志
  5. 注册领域事件（publishAfterCommit）

事务提交后（异步）：
  6. WebSocket 推送
  7. 搜索索引更新
```

### 错误模式

```java
// ❌ 错误：在事务内调用外部服务
@Transactional
public void createAndUpload(...) {
    db.insert(message);
    storageService.upload(file);  // 外部 API 超时，事务不提交！
}

// ❌ 错误：在 Controller 中开启事务
@Transactional  // Controller 层不应该有事务注解
public R<Long> add(...) { ... }
```

### 正确的做法

```java
// ✅ 正确：事务内只做数据库操作
@Transactional
public Long sendMessage(...) {
    db.insert(message);
    db.update(chat);
    eventPublisher.registerEvent("message.created", message);
    return message.getId();
}
```
